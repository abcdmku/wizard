## Integration Examples

### Basic Wizard

```typescript
import { wizardWithContext } from '@wizard/core';

interface MyContext {
  userId: string;
  step_count: number;
}

const { defineSteps, createWizard, step } = wizardWithContext<MyContext>({
  userId: '',
  step_count: 0
});

const steps = defineSteps({
  welcome: step({
    data: { acknowledged: false },
    next: ['profile']
  }),
  profile: step({
    data: { name: '', email: '' },
    canExit: ({ data }) => Boolean(data?.name && data?.email),
    next: ['complete']
  }),
  complete: step({
    data: { timestamp: 0 },
    next: []
  })
});

const wizard = createWizard(steps);
```

### React Integration

```tsx
import { WizardProvider, useWizard, useWizardStep } from '@wizard/react';

function MyWizardComponent() {
  const wizard = useWizard();
  const currentStep = useWizardStep();

  return (
    <div>
      <h1>Current Step: {currentStep}</h1>
      <button onClick={() => wizard.next()}>
        Next Step
      </button>
    </div>
  );
}

function App() {
  return (
    <WizardProvider wizard={wizard}>
      <MyWizardComponent />
    </WizardProvider>
  );
}
```

## Migration Guide

### From Legacy Patterns

If you're migrating from older WizardOpus versions:

```typescript
// Old pattern (deprecated)
const wizard = createWizard({
  context: { /* ... */ },
  steps: defineSteps({ /* ... */ })
});

// New pattern (recommended)
const { defineSteps, createWizard, step } = wizardWithContext({
  // Initial context
});

const steps = defineSteps({
  // Steps with automatic type inference
});

const wizard = createWizard(steps);
```
