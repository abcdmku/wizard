# React Quick Start

## Define a typed wizard

Use `createReactWizardFactory` to define steps with components.

```tsx
import {
  createReactWizardFactory,
  type StepComponentProps,
} from '@wizard/react';

type Context = { total: number };
type StepName = 'account' | 'review';
type WizardData = {
  account: { email: string };
  review: { agreed: boolean };
};

type AccountStepProps = StepComponentProps<
  Context,
  StepName,
  WizardData,
  never,
  'account'
>;

type ReviewStepProps = StepComponentProps<
  Context,
  StepName,
  WizardData,
  never,
  'review'
>;

const factory = createReactWizardFactory<Context>();
const { defineSteps, step } = factory;

function AccountStep({ step, data }: AccountStepProps) {
  return (
    <button onClick={() => step.updateData({ email: 'user@example.com' })}>
      {data?.email ?? 'Set email'}
    </button>
  );
}

function ReviewStep({ wizardData }: ReviewStepProps) {
  return <pre>{JSON.stringify(wizardData, null, 2)}</pre>;
}

const steps = defineSteps({
  account: step({
    data: { email: '' },
    component: AccountStep,
    next: ['review'],
  }),
  review: step({
    data: { agreed: false },
    component: ReviewStep,
    next: [],
  }),
});

export const wizard = factory.createWizard(steps, {
  context: { total: 0 },
});
```

## Bind wizard state in components

Use hooks with a wizard instance:

```tsx
import { useWizard, useWizardStep } from '@wizard/react';

function Screen() {
  const { step } = useWizard(wizard);
  const account = useWizardStep(wizard, 'account');
  return <div>{step} / {account.status}</div>;
}
```

## Optional provider

`WizardProvider` is optional and can be used to avoid passing the wizard argument in every hook call.
