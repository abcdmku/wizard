# Quick Start with React

To use Wizard with React, you'll need the `@wizard/react` package.

## Installation

```bash
npm install @wizard/core @wizard/react
```

## 1. Define Your Wizard

It's best practice to define your wizard logic in a separate file (e.g., `wizard.config.ts`).

```typescript
// wizard.config.ts
import { wizardWithContext } from '@wizard/core';
import type { AppContext } from './types';

// 1. Create factory with context type
const { defineSteps, createWizard, step } = wizardWithContext<AppContext>({
  userId: 'guest',
  theme: 'light'
});

// 2. Define steps
const steps = defineSteps({
  name: step({
    data: { firstName: '', lastName: '' },
    next: ['email']
  }),
  email: step({
    data: { email: '' },
    next: ['confirm']
  }),
  confirm: step({
    next: []
  })
});

// 3. Export the instance
export const wizard = createWizard(steps);
```

## 2. Setup the Provider

Wrap your application (or the specific route) with the `WizardProvider`.

```tsx
// App.tsx
import { WizardProvider } from '@wizard/react';
import { wizard } from './wizard.config';
import { WizardFlow } from './WizardFlow';

export default function App() {
  return (
    <WizardProvider wizard={wizard}>
      <WizardFlow />
    </WizardProvider>
  );
}
```

## 3. Build Your Components

Use the `useWizard` and `useWizardStep` hooks to interact with the state.

```tsx
// WizardFlow.tsx
import { useWizard, useWizardStep } from '@wizard/react';

export function WizardFlow() {
  const { step, next, back } = useWizard();

  return (
    <div className="wizard-container">
      {step === 'name' && <NameStep />}
      {step === 'email' && <EmailStep />}
      {step === 'confirm' && <ConfirmStep />}
    </div>
  );
}

function NameStep() {
  // Use step-specific hook for data and updates
  const { data, setStepData, next } = useWizardStep('name');

  return (
    <div>
      <h2>What is your name?</h2>
      <input 
        value={data.firstName}
        onChange={e => setStepData({ firstName: e.target.value })}
      />
      <button onClick={next}>Next</button>
    </div>
  );
}
```
