# Building UI

## Read current step and progress

```tsx
import { useWizard, useWizardProgress } from '@wizard/react';

function Header() {
  const { step } = useWizard(wizard);
  const progress = useWizardProgress(wizard);

  return (
    <div>
      <div>Current: {step}</div>
      <div>{Math.round(progress.percentage)}%</div>
    </div>
  );
}
```

## Work with a specific step

```tsx
import { useWizardStep } from '@wizard/react';

function AccountForm() {
  const step = useWizardStep(wizard, 'account');

  return (
    <button
      onClick={() => {
        step.updateData({ email: 'user@example.com' });
        void step.next();
      }}
    >
      Save and continue
    </button>
  );
}
```

## Select minimal state for rerender control

```tsx
import { useWizardSelector } from '@wizard/react';

function ErrorBadge() {
  const hasErrors = useWizardSelector(
    wizard,
    (state) => Object.keys(state.errors).length > 0
  );

  return hasErrors ? <span>Errors</span> : null;
}
```
